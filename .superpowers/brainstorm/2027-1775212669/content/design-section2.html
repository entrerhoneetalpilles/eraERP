Modèle de données Prisma + Layout shadcn/ui du back-office

Schéma Prisma — Entités Bloc 1
packages/database/schema.prisma (entités principales)
model User {
  id        String   @id @default(cuid())
  email     String   @unique
  role      Role     // ADMIN_SI | DIRECTION | GESTIONNAIRE | COMPTABLE | ...
  createdAt DateTime @default(now())
}

model Proprietaire {
  id          String  @id @default(cuid())
  type        TypeProprietaire // PARTICULIER | SCI | INDIVISION
  nom         String
  prenom      String?
  email       String
  telephone   String?
  adresse     Json    // rue, cp, ville, pays
  biens       Bien[]
  mandats     Mandat[]
  compteMandant CompteMandant?
  createdAt   DateTime @default(now())
}

model Bien {
  id              String  @id @default(cuid())
  proprietaireId  String
  proprietaire    Proprietaire @relation(...)
  type            TypeBien  // APPARTEMENT | MAISON | STUDIO | ...
  adresse         Json
  surface         Float
  nbPieces        Int
  dpe             String?  // A-G
  loyer           Decimal
  charges         Decimal
  photos          String[]
  mandat          Mandat?
  baux            Bail[]
}

model Mandat {
  id              String   @id @default(cuid())
  numero          String   @unique  // Numérotation Hoguet
  proprietaireId  String
  bienId          String   @unique
  dateDebut       DateTime
  dateFin         DateTime?
  tauxHonoraires  Decimal  // % du loyer
  seuilTravaux    Decimal  // Seuil accord proprio pour devis
  statut          StatutMandat // ACTIF | RESILIÉ | SUSPENDU
  documents       Document[]
}
